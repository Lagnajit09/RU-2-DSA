#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
#define s 100
int stack[s];
int top=-1;
void push()
{
 int n;
 if(top == s-1)
 {
  printf("Overflow\n");
 }
 else{
  top++;
  printf("Enter the value.\n");
  scanf("%d",&n);
  stack[top]=n;
 }
}
void pop()
{
 if(top == -1)
 {
  printf("Underflow.\n");
 }
 else{
  printf("%d\n",stack[top]);
  top--;
 }
}
void traverse()
{
 int i;
 if(top == -1)
 {
  printf("Underflow\n");
 }
 for(i=0; i<=top; i++)
 {
  printf("%d,",stack[i]);
 }
}
void main()
{
 int ch,i,n;
 clrscr();
 for(i=0; i<s; i++)
 {
  printf("\n\nEnter\n1 for push\n2 for pop\n3 for traverse\n4 to exit\n");
  scanf("%d",&ch);
  switch(ch)
  {
   case 1:
    push();
    break;
   case 2:
    pop();
    break;
   case 3:
    traverse();
    break;
   case 4:
    exit(0);
   default:
    printf("Invalid Entry.\n");
  }
 }
 getch();
}
